<Note>
  **Prerequisites:** A Solana wallet address to receive settlement payouts.
</Note>

<Info>
  Examples below use `https://api-devnet.agonx402.com` (devnet). For production, use `https://api-mainnet.agonx402.com`. See [Environments](/api-reference/introduction#environments).
</Info>

<Steps>
  <Step title="Install & Register">
    ```bash
    npm install @agonx402/platform @agonx402/types
    ```

    ```bash
    curl -X POST https://api-devnet.agonx402.com/platform/register \
      -H "Content-Type: application/json" \
      -d '{"name": "My API", "wallet_address": "YourSolanaAddress"}'
    ```

    Save the returned `platform_key` as `AGON_PLATFORM_KEY` in your environment.
  </Step>

  <Step title="Protect Your Endpoints">
    <Tabs>
      <Tab title="Fastify">
        ```typescript
        import Fastify from 'fastify'
        import { AgonPlatformCore, generateRequestId } from '@agonx402/platform'

        const app = Fastify()
        const agon = new AgonPlatformCore({
          agonUrl: 'https://api-devnet.agonx402.com',
          platformKey: process.env.AGON_PLATFORM_KEY!,
          pricing: 1000, // 1000 micro-USDC = $0.001 per call
        })

        app.addHook('preHandler', async (req, reply) => {
          const token = agon.extractConsumerToken(req.headers)
          if (!token) {
            const { status, body } = agon.buildPaymentRequiredResponse(BigInt(1000))
            return reply.status(status).send(body)
          }
          const auth = await agon.authorize(token, generateRequestId(), BigInt(1000))
          if (auth.status === 'denied') return reply.status(402).send({ reason: auth.reason })
          req.reservationId = auth.reservation_id!
          reply.raw.on('finish', () => {
            if (reply.statusCode < 400) agon.consume(req.reservationId)
            else agon.release(req.reservationId)
          })
        })

        app.get('/api/data', async () => ({ result: 'premium data' }))
        app.listen({ port: 3000 })
        ```
      </Tab>

        app.use('/api/v1', async (req, res, next) => {

          if (!token) {
            const { status, body } = agon.buildPaymentRequiredResponse(BigInt(1000))
            return res.status(status).json(body)
          }
          const auth = await agon.authorize(token, generateRequestId(), BigInt(1000))
          if (auth.status === 'denied') return res.status(402).json({ reason: auth.reason })
          req.reservationId = auth.reservation_id!
          res.on('finish', () => {
            if (res.statusCode < 400) agon.consume(req.reservationId)
            else agon.release(req.reservationId)
          })
          next()
        })

        app.get('/api/v1/data', (req, res) => res.json({ result: 'premium data' }))
        app.listen(3000)

      <Tab title="Next.js">
        ```typescript
        // app/api/data/route.ts
        import { AgonPlatformCore, generateRequestId } from '@agonx402/platform'
        import { NextRequest, NextResponse } from 'next/server'

        const agon = new AgonPlatformCore({
          agonUrl: 'https://api-devnet.agonx402.com',
          platformKey: process.env.AGON_PLATFORM_KEY!,
          pricing: 1000,
        })

        export async function GET(req: NextRequest) {
          const token = agon.extractConsumerToken(Object.fromEntries(req.headers))
          if (!token) {
            const { status, body } = agon.buildPaymentRequiredResponse(BigInt(1000))
            return NextResponse.json(body, { status })
          }
          const auth = await agon.authorize(token, generateRequestId(), BigInt(1000))
          if (auth.status === 'denied') return NextResponse.json({ reason: auth.reason }, { status: 402 })
          const response = NextResponse.json({ result: 'premium data' })
          agon.consume(auth.reservation_id!).catch(console.error)
          return response
        }
        ```

  <Step title="Collect Earnings">
    When you're ready to settle, trigger it via the Agon dashboard or API:

    ```bash
    curl -X POST https://api-devnet.agonx402.com/platform/settle \
      -H "Authorization: Bearer pk_xxxxx"
    ```

    All consumed charges are aggregated per consumer and settled to your wallet in a single on-chain USDC transfer.
  </Step>
